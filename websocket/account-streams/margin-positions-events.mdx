Subscribe to margin call and liquidation event notifications for collateral positions. Events fire when a position enters margin call or liquidation state.

<WsSubscribeSteps
  subscribe={exMarginPositionsEventsSubscribe}
  subscribeNote="params is always an empty array."
  confirmation={exMarginPositionsEventsSubscribeResponse}
  update={exMarginPositionsEventsUpdate}
  updateNote="params[0] is the event type. params[1] is the position object."
  updateExtra={
    <div>
      <WsTupleTable fields={marginPositionsEventsUpdateParamsTupleFields} />

      </div>
    </div>
  }
  unsubscribe={exMarginPositionsEventsUnsubscribe}
/>
